43 {
44  size_t size = snprintf(nullptr, 0, format.c_str(), args ...) + 1;
45  if (size <= 0) { // LCOV_EXCL_LINE
46  throw std::runtime_error("Error during formatting."); // LCOV_EXCL_LINE
47  }
48  std::unique_ptr<char[]> buf(new char[size]);
49  snprintf(buf.get(), size, format.c_str(), args ...);
